def norm(x):
    return (x-np.mean(x))/np.std(x)

n_samples=5000

t=np.arange(n_samples)
x1=stats.norm.rvs(loc=0,scale=1,size=n_samples)

x3=stats.norm.rvs(loc=0,scale=1,size=n_samples)

X=np.stack([norm(x1),norm(x2),norm(x3)],axis=1);
